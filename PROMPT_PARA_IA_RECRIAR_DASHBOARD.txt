```typescript
export function formatDate(date: Date | string): string {
  const dateObj = typeof date === 'string' ? new Date(date) : date;
  return dateObj.toLocaleDateString('pt-BR');
}

export function formatRelativeTime(date: Date | string): string {
  const dateObj = typeof date === 'string' ? new Date(date) : date;
  const now = new Date();
  const diff = now.getTime() - dateObj.getTime();
  
  const seconds = Math.floor(diff / 1000);
  const minutes = Math.floor(seconds / 60);
  const hours = Math.floor(minutes / 60);
  const days = Math.floor(hours / 24);
  
  if (days > 0) return `h√° ${days} dia${days > 1 ? 's' : ''}`;
  if (hours > 0) return `h√° ${hours} hora${hours > 1 ? 's' : ''}`;
  if (minutes > 0) return `h√° ${minutes} minuto${minutes > 1 ? 's' : ''}`;
  return `h√° ${seconds} segundo${seconds > 1 ? 's' : ''}`;
}
```

```typescript
interface LanguageContextType {
  language: string;
  setLanguage: (lang: string) => void;
  t: (key: string) => string;

const LanguageContext = createContext<LanguageContextType>({
  language: 'pt-BR',
  setLanguage: () => {},
  t: (key: string) => key,
});

const translations: Record<string, Record<string, string>> = {
  'pt-BR': {
    'dashboard.welcomeBack': 'Bem-vindo de volta',

    'dashboard.cashShop': 'Cash Shop',
    'dashboard.vipBenefits': 'Benef√≠cios VIP',
  },
};

export function LanguageProvider({ children }: { children: React.ReactNode }) {
  const [language, setLanguage] = useState('pt-BR');
  
  const t = (key: string) => translations[language]?.[key] || key;
  
  return (
    <LanguageContext.Provider value={{ language, setLanguage, t }}>
```
